```cpp
class Solution {
public:
    int firstUniqChar(string s) {
        unordered_map<char, int> count;

        // Step 1: Count frequency
        for (char ch : s) {
            count[ch]++;
        }

        // Step 2: Find the first character with count == 1
        for (int i = 0; i < s.length(); i++) {
            if (count[s[i]] == 1) {
                return i;
            }
        }

        return -1;
    }
};
```
