/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
public:
    void solve(TreeNode* node, int depth, map<int, int>& levelEle) {
        if (node == NULL)
            return;

        // if (levelEle.find(depth) != levelEle.end()) {
            levelEle[depth] = node -> val;
        // }

        solve(node->left, depth + 1, levelEle);
        solve(node->right, depth + 1, levelEle);
    }
    vector<int> rightSideView(TreeNode* root) {

        int depth = 0;
        map<int, int> levelEle;
        solve(root, depth, levelEle);
        vector<int> ans;
        for (auto i : levelEle) {
            ans.push_back(i.second);
        }
        return ans;
    }
};
