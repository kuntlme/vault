/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void solve(TreeNode* node, int level, int ind, map<int, map<int, vector<int>>> &levelEle){
        if(node == NULL){
            return;
        }

            // level exist
        if(levelEle.find(level) != levelEle.end()){
            //index exist
            if(levelEle[level].find(ind) != levelEle[level].end()){
                levelEle[level][ind].push_back(node -> val);
            }
            // index not exist
            else{
            vector<int>temp;
            temp.push_back(node -> val);
            levelEle[level][ind] = temp;
            }
        }
        else{
            vector<int> temp;
            temp.push_back(node -> val);
            levelEle[level][ind] = temp;
        }

        solve(node -> left, level - 1, ind + 1, levelEle);
        solve(node -> right, level + 1, ind + 1, levelEle);

    }
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        int level = 0;
        int ind = 0;
        map<int, map<int, vector<int>>> levelEle;
        solve(root, level, ind, levelEle);
        vector<vector<int>>ans;
        for(auto i: levelEle){
            vector<int>temp;
            for(auto j: i.second){
              sort(j.second.begin(), j.second.end());
              for(auto ind: j.second){
                temp.push_back(ind);
              }
            }
            ans.push_back(temp);
        }
        return ans;
    }
};
